#include <iostream>
#include<bits/stdc++.h>
using namespace std;

void swap(int &a, int &b){
    int t = a;
    a = b;
    b = t;
}

int main() {
    // Selection Sort

    int n;
    cin>>n;

    int arr[n];

    for(int i=0;i<n;i++){
        cin>>arr[i];
    }

    for(int i=1;i<n;i++){
        int unsortedElement = arr[i], j = i-1;
        while(j>=0 && unsortedElement < arr[j]){
            arr[j+1] = arr[j];
            j--;
        }        
        arr[j+1] = unsortedElement;
    }

    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }

    return 0;
}
